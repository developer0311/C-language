//Write a C program to design calculator with basic operation using switch case.

#include<stdio.h>
int main(void)
{
     char operation;
     float a,b,c;
     printf("\nEnter what operation you want do ['+','-','*','/'] : ");
     scanf("%c",&operation);
     printf("\nEnter First operand : ");
     scanf("%f",&a);
     printf("Enter Second operand : ");
     scanf("%f",&b);
     switch(operation)
     {
          case '+':
               c=a+b;
               break;
          case '-' :
               c=a-b;
               break;
          case '*':
               c=a*b;
               break;
          case '/':
               c=a/b;
               break;
          default :
               printf("\nEnter Valid Inputs .");
     }
     printf("\n%0.2f %c %0.2f : %0.2f",a,operation,b,c);
     return 0;
}

//Write a C program to create a chart of factorial of 1 to 5 .

#include<stdio.h>
int main (void)
{
     int i=1,j,fact=1;
     while(i<=5)
     {
          for(j=1;j<=i;j++)
               fact*=j;
          printf("\nFactorial of %d is : %d",i,fact);
          fact=1;
          i++;
     }
    
     return 0;
}
